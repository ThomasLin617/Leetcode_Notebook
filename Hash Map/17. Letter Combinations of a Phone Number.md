# 17. Letter Combinations of a Phone Number
## code
```python
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        dic = {
            '2' : 'abc',
            '3' : 'def',
            '4' : 'ghi',
            '5' : 'jkl',
            '6' : 'mno',
            '7' : 'pqrs',
            '8' : 'tuv',
            '9' : 'wxyz',


        }
        if not digits:
            return []
        
        def backtrack(indx, comb):
            if indx == len(digits):
                res.append(comb[:])
                return
            for letter in dic[digits[indx]]:
                backtrack(indx+1, comb+letter)

        res = []
        backtrack(0, "")
        return res
```
